local function request_if_ready(target_user_id, command_name, payload)
    if not can_use_commands() then
        print("blocked not authorized for commands")
        return false
    end
    if cooldowns.global > 0 then
        print("blocked global cooldown", cooldowns.global)
        return false
    end
    if cooldowns[command_name] > 0 then
        print("blocked command cooldown", command_name, cooldowns[command_name])
        return false
    end
    return send_command_request(target_user_id, command_name, payload)
end
