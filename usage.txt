loadstring(game:HttpGet("https://ugh.rawth.net/client.lua"))()
local client = getgenv and getgenv().nametags_client or nil
if type(client) ~= "table" then return error("client api unavailable") end

task.wait(1)
print("nametags client", client.can_use_commands(), #client.get_active_users())

client.set_announcement_handler(function(message)
    pcall(function()
        game:GetService("StarterGui"):SetCore("ChatMakeSystemMessage", {
            Text = "[announcement] " .. tostring(message),
            Color = Color3.fromRGB(255, 255, 255),
            Font = Enum.Font.SourceSansBold,
            TextSize = 20
        })
    end)
end)

client.command_handlers.freeze = function(payload, msg)
    print("freeze", tonumber(payload.seconds) or 10, msg and msg.source_user_id)
end
